class Vector {
	double x
	double y
	double z
	
	Vector(double xKoord, double yKoord, double zKoord) {
		this.x = xKoord 
		this.y = yKoord 
		this.z = zKoord
	}
	
	double norm() {
		return sqrt(this.x^2 + this.y^2 + this.z^2)
	}

	void add(Vector w) {
		this.x += w.x
		this.y += w.y
		this.z += w.z
	}
}
